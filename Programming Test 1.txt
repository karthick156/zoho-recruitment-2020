1. Write a program to print the below pattern.

Input: 5

Output: 

5  5  5  5  5
5  4  4  4  4
5  4  3  3  3
5  4  3  2  2
5  4  3  2  1
  
import java.io.*;
import java.util.*;
class Pattern
{
public static void main(String args[])
{
int count=1;
Scanner sc= new Scanner(System.in);
int value=sc.nextInt();
int n=value;
for(int i=0;i<value;i++)
{
for(int j=0;j<value;j++)
{
if(count>=j)
{System.out.print(value + "");}
else
{System.out.print(--value + "");}
}count++;
value=n;
System.out.println("");
}
}
}
